data("ex_mcfa1")
sym.table <- classic.to.sym(ex_mcfa1,
                            concept = suspect,
                            hair = sym.set(hair),
                            eyes = sym.set(eyes),
                            region = sym.set(region))

mcfa.scatterplot(res[,1], res[,2], sym.data = sym.table, pos.var = c(1,2))
